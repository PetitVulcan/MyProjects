CREATE TABLE [dbo].[utilisateur] (
    [Id]             INT          IDENTITY (1, 1) NOT NULL,
    [date]           DATETIME     NULL,
    [nom]            VARCHAR (50) NULL,
    [prenom]         VARCHAR (50) NULL,
    [nomutilisateur] VARCHAR (50) NULL,
    [email]          VARCHAR (50) NULL,
    [nbrecettecree]  INT          NULL,
    [mdp]            VARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[recette] (
    [Id]             INT          IDENTITY (1, 1) NOT NULL,
    [date]           DATETIME     NULL,
    [nomutilisateur] VARCHAR (20) NULL,
    [nomrecette]     VARCHAR (50) NULL,
    [nbpersonne]     INT          NULL,
    [difficulte]     VARCHAR (5)  NULL,
    [ingredient]     NTEXT        NULL,
    [realisation]    NTEXT        NULL,
    [IdCategory]     INT          NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[nbpersonne] (
    [Id]          INT IDENTITY (1, 1) NOT NULL,
    [nbpersonnes] INT NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[images] (
    [Id]        INT           IDENTITY (1, 1) NOT NULL,
    [Idrecette] INT           NOT NULL,
    [urlimage]  VARCHAR (MAX) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[difficulte] (
    [Id]    INT          IDENTITY (1, 1) NOT NULL,
    [titre] VARCHAR (20) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Category] (
    [Id]    INT          IDENTITY (1, 1) NOT NULL,
    [Titre] VARCHAR (20) NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
